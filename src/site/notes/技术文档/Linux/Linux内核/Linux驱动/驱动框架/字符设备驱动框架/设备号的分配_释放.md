# 申请方式
## 动态申请
**步骤**
1. 模块初始化时统一申请
2. 定义次设备分配器, `eg. static DEFINE_IDA(minor_ida);`
3. `probe`中调用[[技术文档/Linux/Linux内核/源码结构/include/linux/idr.h#ida_alloc()\|idr.h#ida_alloc()]]动态分配子设备编号

**释放**
1. [[技术文档/Linux/Linux内核/源码结构/include/linux/idr.h#ida_free()\|idr.h#ida_free()]]

**示例**
```c
/* ----- 方法一 ----- */
#include <linux/idr.h>

#define MAX_DEVICES 8
static dev_t dev_base;             // 起始设备号
static struct class *drv_class;
static DEFINE_IDA(minor_ida);      // 次设备号分配器

static int my_probe(struct i2c_client *client, const struct i2c_device_id *id)
{
    struct my_dev_data *data;
    int minor, ret;
    dev_t dev_no;

    minor = ida_alloc(&minor_ida, GFP_KERNEL);
    if (minor < 0)
        return minor;

    data = devm_kzalloc(&client->dev, sizeof(*data), GFP_KERNEL);
    if (!data) {
        ret = -ENOMEM;
        goto err_ida;
    }
    data->client = client;
    data->dev_num = minor;

    dev_no = MKDEV(MAJOR(dev_base), minor);
    cdev_init(&data->cdev, &my_fops);
    data->cdev.owner = THIS_MODULE;
    ret = cdev_add(&data->cdev, dev_no, 1);
    if (ret)
        goto err_free;

    device_create(drv_class, &client->dev, dev_no, NULL, "mydev%d", minor);
    i2c_set_clientdata(client, data);
    dev_info(&client->dev, "Registered as /dev/mydev%d\n", minor);
    return 0;

err_free:
    devm_kfree(&client->dev, data);
err_ida:
    ida_free(&minor_ida, minor);
    return ret;
}

static void my_remove(struct i2c_client *client)
{
    struct my_dev_data *data = i2c_get_clientdata(client);
    dev_t dev_no = MKDEV(MAJOR(dev_base), data->dev_num);
    device_destroy(drv_class, dev_no);
    cdev_del(&data->cdev);
    ida_free(&minor_ida, data->dev_num);
    dev_info(&client->dev, "Removed\n");
}

static const struct i2c_device_id my_id_table[] = { { "my_dev", 0 }, {} };
MODULE_DEVICE_TABLE(i2c, my_id_table);

static struct i2c_driver my_driver = {
    .driver = { .name = DRIVER_NAME },
    .probe  = my_probe,
    .remove = my_remove,
    .id_table = my_id_table,
};

static int __init my_init(void)
{
    int ret;
    ret = alloc_chrdev_region(&dev_base, 0, MAX_DEVICES, DRIVER_NAME);
    if (ret)
        return ret;
    drv_class = class_create(THIS_MODULE, DRIVER_NAME);
    if (IS_ERR(drv_class)) {
        ret = PTR_ERR(drv_class);
        goto err_unreg;
    }
    ret = i2c_add_driver(&my_driver);
    if (ret)
        goto err_class;
    pr_info("Method1 loaded, major=%d\n", MAJOR(dev_base));
    return 0;
err_class:
    class_destroy(drv_class);
err_unreg:
    unregister_chrdev_region(dev_base, MAX_DEVICES);
    return ret;
}

static void __exit my_exit(void)
{
    i2c_del_driver(&my_driver);
    class_destroy(drv_class);
    unregister_chrdev_region(dev_base, MAX_DEVICES);
    ida_destroy(&minor_ida);
}

module_init(my_init);
module_exit(my_exit);
MODULE_LICENSE("GPL");
```

# 区别
|方式|主设备号|次设备号管理|cdev 注册次数|支持热插拔|代码复杂度|推荐度|
|---|---|---|---|---|---|---|
|**方式一（池+IDA）**|统一|动态分配|每个设备一次|✅|中等|⭐⭐⭐⭐⭐ (最通用)|
|**方式二（单cdev覆盖范围）**|统一|硬编码偏移|全局一次|❌（数量固定）|最低|⭐⭐⭐⭐ (数量固定时)|
|**方式三（各自申请）**|**不统一**|各自从0开始|每个设备一次|✅|最低|⭐ (强烈不推荐)|
|**方式四（固定主设备号）**|固定|同方式一/二|视情况|视情况|中等|⭐⭐⭐ (特殊场景)|
